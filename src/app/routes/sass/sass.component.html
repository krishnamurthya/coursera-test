Style with @if and @else
CSS
@mixin border-stroke($val) {
@if $val == light { border: 1px solid black; }
@else if $val == medium { border: 3px solid black; }
@else if $val == heavy { border: 6px solid black; }
@else { border: none; }
}
#box { @include border-stroke(medium); }
Output
<div id="box"></div>
Style with @each
CSS
$colors: (color1: blue, color2: black, color3: red);
@each $key, $color in $colors {
.#{$color}-bg {background-color: $color;height: 200px; width: 200px;} }

<div class="blue-bg"></div>
<div class="black-bg"></div>
<div class="red-bg"></div>
